Q1
CREATE TABLE EMPLOYEE (
EMPNO		NUMBER(4)	CONSTRAINT EMPLOYEE_EMPNO_PK	PRIMARY KEY,
NAME		VARCHAR2(20)	NOT NULL,
DESIGNATION	VARCHAR2(20)	NOT NULL,
QUALIFICATION	VARCHAR2(9)	NOT NULL,
JOINDATE 	DATE		DEFAULT SYSDATE,
CONSTRAINT EMPLOYEE_DESIGNATION_FK 	FOREIGN KEY(DESIGNATION) REFERENCES GRADE(DESIGNATION)
);


CREATE TABLE GRADE(
DESIGNATION	VARCHAR2(30)	CONSTRAINT GRADE_DESIGNATION_PK	PRIMARY KEY,
GRADE		NUMBER(2),
TOTALPOSTS	NUMBER(2),	
POSTSAVAILABLE	NUMBER(2),	
CONSTRAINT GRADE_GRADE_CK	CHECK(GRADE BETWEEN 1 AND 20),
CONSTRAINT GRADE_POSTSAVAILABLE_CK	CHECK(POSTSAVAILABLE <= TOTALPOSTS)
);


CREATE TABLE PROJECT(
PID		NUMBER(4)	CONSTRAINT PROJECT_PID_PK	PRIMARY KEY,
TITLE		VARCHAR2(20),
CLIENT		VARCHAR2(30),
DURATION(WEEKS)	NUMBER(2),
STATUS		VARCHAR2(11),
CHECK(STATUS IN('NEW','IN PROGESS','COMPLETE')),
CONSTRAINT PROJECT_TITLE_UK  UNIQUE(TITLE)
);


CREATE TABLE TRAINING(
TCODE		NUMBER(4)	CONSTRAINT TRAINING_TCODE_PK	PRIMARY KEY,
TITLE		VARCHAR2(20),
STARTDATE	DATE,
ENDDATE		DATE,
CONSTRAINT TRAINING_TITLE_UK  UNIQUE(TITLE)
);


CREATE TABLE EMP_PROJECT(
EMPNO		NUMBER(4),	
PID		NUMBER(4),
PERFORMANCE	VARCHAR2(10),
CONSTRAINT EMP_PROJECT_PERFORMANCE_CK CHECK(PERFORMANCE IN('EXCELLENT','GOOD','FAIR','POOR','BAD')),
CONSTRAINT EMP_PROJECT_EMPLOYEE_FK FOREIGN KEY(EMPNO) REFERENCES EMPLOYEE(EMPNO),
CONSTRAINT EMP_PROJECT_PROJECT_FK FOREIGN KEY(PID) REFERENCES PROJECT(PID)
);


CREATE TABLE EMP_TRAINING(
EMPNO		NUMBER(4),
TCODE		NUMBER(4),
ATTENDANCE	NUMBER(3),
CONSTRAINT EMP_TRAINING_ATTENDANCE_CK	CHECK(ATTENDANCE BETWEEN  0 AND 100),
CONSTRAINT EMP_TRAINING_EMPNO_FK	FOREIGN KEY(EMPNO) REFERENCES EMPLOYEE(EMPNO),
CONSTRAINT EMP_TRAINING_TCODE_FK	FOREIGN KEY(TCODE) REFERENCES TRAINING(TCODE)
);
COMMIT;

CREATE VIEW DESVU AS
SELECT DESIGNATION, COUNT (*) "NO_OF_EMP" FROM EMPLOYEE
GROUP BY DESIGNATION;

CREATE VIEW PERVU AS
SELECT E.ENAME,EP.EMPNO,EP.PERFORMANCE,P.TITLE
FROM EMPLOYEE E, PROJECT P, EMP_PROJECT EP
WHERE E.EMPNO=EP.EMPNO AND P.PID=EP.PID;

CREATE VIEW EMPVU AS 
SELECT EP.EMPNO,E.ENAME,EP.COUNT(*) "NO_OF_PROJECTS"
FROM EMPLOYEE E, EMP_PROJECT EP
WHERE EP.EMPNO=E.EMPNO AND EP.PERFORMANCE='EXCELLENT';
