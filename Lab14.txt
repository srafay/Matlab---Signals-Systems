Q1.
SET SERVEROUTPUT ON
ACCEPT
	P_SAL PROMPT 'Enteeer Salary'
DECLARE
	V_ENAME EMP.ENAME%TYPE;
	V_SAL EMP.SAL%TYPE='&P_SAL';
BEGIN
	SELECT ENAME INTO V_ENAME 
	FROM EMP
	WHERE SAL=V_SAL;
	DBMS_OUTPUT.PUT_LINE('No employee with a salary of'||V_SAL);
EXCEPTION:
	WHEN NO_DATA_FOUND THEN
		DBMS_OUTPUT.PUT_LINE('No employee with a salary of'||V_SAL);
	WHEN TOO_MANY_ROWS THEN
		DBMS_OUTPUT.PUT_LINE('More than one employee with salary'||);
	WHEN OTHERS THEN
		DBMS_OUTPUT.PUT_LINE('Some other error occured');

Q2.
SET SERVEROUTPUT ON
DECLARE
	V_EMPNO EMP.EMPNO%TYPE;
	V_SAL EMP.SAL%TYPE;
	MSAL100 EXCEPTION;
	PSAL100 EXCEPTION;
	PRAGMA EXCEPTION_INIT(MSAL100);
	PRAGMA EXCEPTION_INIT(PSAL100);
BEGIN 
	SELECT EMPNO INTO V_EMPNO 
	FROM EMP
	WHERE SAL=V_SAL;
	IF V_SAL=SAL+100 THEN
		RAISE PSAL100;
	ELSEIF V_SAL=SAL-100 THEN
		RAISE MSAL100;
	ELSE
		RAISE NO_DATA_FOUND;
	END IF;
	EXCEPTION:
	WHEN PSAL100 THEN
		DBMS_OUTPUT.PUT_LINE('+100 SAL'||V_EMPNO);
	WHEN MSAL100 THEN
		DBMS_OUTPUT.PUT_LINE('-100 SAL'||V_EMPNO);
	WHEN TOO_MANY_ROWS THEN
		DBMS_OUTPUT.PUT_LINE('Too many employees are manager'||COUNT(V_EMPNO));
	WHEN OTHERS THEN
		DBMS_OUTPUT.PUT_LINE('other error');
END;

Q3.
SET SERVEROUTPUT ON
ACCEPT
	P_EMPNAME PROMPT 'Enter employee name');
DECLARE
	V_EMPNAME EMP.ENAME%TYPE='&P_EMPNAME';
	V_SAL EMP.SAL%TYPE;
	V_COMM EMP.COMM%TYPE;
	V_DESG EMP.JOB%TYPE;
	P_NET NUMBER=0;
BEGIN
	SELECTSAL,COMM,JOB INTO V_SAL,V_COMM,V_DESG FROM EMP WHERE ENAME=V_EMPNAME;
	IF V_DESG='SALESMAN' THEN
		P_NET=V_SAL/V_COMM;
		DBMS_OUTPUT.PUT_LINE('Result is:'||P_NET);
	ELSE
		DBMS_OUTPUT.PUT_LINE('Not a salesman '||V_EMPNAME);
	END IF;
EXCEPTION
	WHEN NO_DATA_FOUND THEN
		DBMS_OUTPUT.PUT_LINE('No record');
	WHEN TOO_MANY_ROWS THEN
		DBMS_OUTPUT.PUT_LINE('Too many employees);
	WHEN ZERO_DIVIDE THEN
		DBMS_OUTPUT.PUT_LINE('No commission for '||V_EMPNAME);
END;
