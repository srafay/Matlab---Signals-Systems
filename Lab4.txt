Q1
SELECT SYSDATE AS "CURRENT DATE" FROM DUAL;

Q2 
SELECT EMPNO,ENAME,SAL,ROUND(SAL*1.15) AS "NEWSALARY", (ROUND(SAL*1.15-SAL)) AS "INCREMENT" FROM EMP;

Q3
SELECT ENAME, ROUND(MONTHS_BETWEEN(SYSDATE,HIREDATE)) AS "MONTHS_WORKED"
FROM EMP
ORDER BY MONTHS_WORKED;

Q4
SELECT ENAME||' EARNS '||SAL||' MONTHLY' FROM EMP;

Q5
SELECT INITCAP(ENAME) AS "NAME",LENGTH(ENAME) AS "LENGTH"
FROM EMP
WHERE ENAME LIKE 'J%' OR ENAME LIKE 'A%' OR ENAME LIKE 'M%';

Q6
SELECT ENAME, HIREDATE, TO_CHAR(HIREDATE,'D') AS "DAY"
FROM EMP
ORDER BY MOD(TO_CHAR(HIREDATE,'D')+5,7);

Q7
SELECT DEPTNO,JOB,COUNT(EMPNO) AS "NUM_EMP"
FROM EMP
GROUP BY DEPTNO,JOB;

Q8
SELECT D.DNAME "DNAME" ,D.LOC "LOC" ,E.COUNT(*) "NUMBER OF PEOPLE",
ROUND(AVG(E.SAL),2) "SALARY"
FROM EMP E,DEPT D
WHERE E.DEPTNO=D.DEPTNO
GROUP BY D.DNAME,D.LOC;

Q9
SELECT E.ENAME,E.DEPTNO,E.JOB
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND D.LOC='DALLAS';

Q10
SELECT MAX(SAL)-MIN(SAL) "DIFFERENCE"
FROM EMP;

Q11
SELECT M.ENAME "MANAGER" ,COUNT(E.MGR) "NUM"
FROM EMP E, EMP M
WHERE E.MGR=M.EMPNO
GROUP BY M.ENAME;
