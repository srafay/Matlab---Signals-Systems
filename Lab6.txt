FROM EMP
WHERE HIREDATE BETWEEN TO_DATE('&DATE1','MM/DD/YYYY') AND TO_DATE('&DATE2','MM/DD/YYYY');

Q2:
COLUMN E.ENAME HEADING 'EMPLOYEE|NAME'
COLUMN D.DNAME HEADING 'DEPARTMENT|NAME'
COLUMN E.HIREDATE HEADING 'START|DATE'
COLUMN E.SAL HEADING 'SALARY'
BREAK ON D.DNAME
REM
SELECT E.ENAME,D.DNAME,D.LOC,E.SAL,E.SAL*12,E.HIREDATE 
FROM EMP E,DEPT D
WHERE E.DEPTNO=D.DEPTNO AND D.LOC='&LOCATION'
ORDER BY D.DNAME;
